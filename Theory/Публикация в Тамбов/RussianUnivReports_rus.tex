\section{Основные понятия}

Пусть задано частично упорядоченное пространство $X=(X, {\preceq}_{_{_X}}).$
Для элементов $x',x \in X,$ удовлетворяющих этому отношению, наряду с
обозначением $x' {\preceq}_{_{_X}} x$
используем также обозначение $x {\succeq}_{_{_X}} x'.$ Если
$x' {\preceq}_{_{_X}} x$ и $x' \neq x,$ пишем $x' {\prec}_{_{_X}} x$
или  $x{\succ}_{_{_X}} x' .$
Для элементов  $u,v \in X$ определим множества
$$\mathcal{O}_X (u) \doteq \{x\in X:\, x{\preceq}_{_{_X}} u \}, \ \
 {[v,u]}_X \doteq \{  x \in X : \  v \preceq x \preceq u \}.$$
Заметим, что ${[v,u]}_X \neq \emptyset $ тогда и только тогда,
когда $v \preceq u .$


 Пусть также задано непустое множество $Y,$ на котором определено бинарное
 отношение ${\preceq}_{_{_Y}},$ являющееся рефлексивным (то есть для любого
 $ y\in Y $ выполнено $y {\preceq}_{_{_Y}} y $). Отношение ${\preceq}_{_{_Y}}$
 не предполагается ни антисимметричным, ни транзитивным. В пространстве $Y$ также пользуемся
 равносильными обозначениями $y' {\preceq}_{_{_Y}} y$ $\Leftrightarrow$ $y {\succeq}_{_{_Y}} y'$
 и  $y' \neq y,$  $y' {\preceq}_{_{_Y}} y$ $\Leftrightarrow$  $y' {\prec}_{_{_Y}} y$
  $\Leftrightarrow$  $y {\succ}_{_{_Y}} y'.$

\begin{Definition}\cite[Definition 1]{1} Отображение $f\!:X\!\to\! Y$ называется \textit{антитонным на мно\-жест\-ве}
 $U\subset X,$ если для любых  $x_1,x_2 \in U$ таких, что $x_1 {\succeq}_{_{_X}} x_2 ,$
 выполнено $$f(x_1) {\preceq}_{_{_Y}} f(x_2).$$
\end{Definition}
\begin{Definition}\cite[с. 475]{2}  Отображение $f:X\to Y$ называется
 (\textit{упорядоченно})~\textit{на\-кры\-ва\-ю\-щим множество}
 $W \subset Y,$ если
\begin{equation}\label{Sarra-Def-1}
 \forall x_{0}\in X \quad \forall y\in W \quad
 y {\preceq}_{_{_Y}} f(x_0) \ \Rightarrow \ \exists x\in X \
 f(x)=y \ \, \mbox{и} \ \, x {\preceq}_{_{_X}} x_0 .
\end{equation}
 \end{Definition}
