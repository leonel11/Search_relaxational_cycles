\begin{frame}
\frametitle{Исследуемая динамическая система} 
\begin{eqnarray}\label{u_system} 
	\dot{u_j} = d(u_{j+1}-2u_j+u_{j-1})+\lambda[-1+\alpha f(u_j(t-1)) - \beta g(u_j)]u_j, \nonumber \\ j = \overline{1,m},
\end{eqnarray}
\pause
$$ \underline{u_0=u_m, \; u_1=u_{m+1}}, $$

$$ u_j=u_j(t)>0, \;  m \ge 2, \; \lambda >> 1, \; \beta > 0, \; \alpha > 1 + \beta, $$
$$ f(u), g(u) \in C^2(\mathbb{R}_+): \quad \mathbb{R}_+  = \{ u \in \mathbb{R}: u \geqslant 0\}, $$
$$ 0 < \beta g(u) < \alpha, \; f(0) = g(0) = 1, \quad \forall u \in \mathbb{R}_+; $$
$$ f(u), g(u), uf'(u), ug'(u), u^2f''(u), u^2g''(u) = O(1/u), \; u \to +\infty. $$
\end{frame}

\begin{frame}
\frametitle{Замены} 
$$ u_1 = \mbox{exp} \left( \frac{x}{\varepsilon} \right), \quad \varepsilon = \frac{1}{\lambda} << 1, $$
$$ u_j = \mbox{exp} \left( \frac{x}{\varepsilon} + \sum\limits_{k=1}^{j-1} y_k \right), \quad j = \overline{2,m}, $$ 
\vspace{1.5cm}
\pause
$$ \dot{x} = -1 + \alpha f \left( \mbox{exp} \, \left( \frac{x(t-1)}{\varepsilon} \right) \right) - \beta g \left( \mbox{exp} \, \left( \frac{x}{\varepsilon} \right) \right). $$
\end{frame}

\begin{frame}
\frametitle{Линейная импульсная система} 
\begin{eqnarray}\label{y_system} 
	\dot{y_j} = d[\mbox{exp} \, y_{j+1} + \mbox{exp}(-y_j) - \mbox{exp} \, y_j - \mbox{exp}(-y_{j-1})], \nonumber \\ j = \overline{1,m-1},
\end{eqnarray}

$$ y_j(+0) = \frac{\alpha -1}{\alpha - \beta - 1}y_j(-0), $$
$$ y_j(1+0) = y_j(1-0) - \frac{\alpha}{\alpha - 1}y_j(+0), $$
$$ y_j(\alpha + 0) = (1 + \beta)y_j(\alpha - 0), $$
$$ y_j(\alpha + 1 + 0) = y_j(\alpha + 1 - 0) - \frac{\alpha}{1 + \beta}y_j(\alpha + 0),  $$
$$ y_0 = y_m = -\sum\limits_{k=1}^{m-1} y_k.  $$
\end{frame}

\begin{frame}
\frametitle{Модельное отображение} 
\begin{equation}\label{phi_z} 
	\Phi(z): \begin{pmatrix}
           z_1 \\
           \vdots \\
           z_{m-1}
          \end{pmatrix}
					\to
					\begin{pmatrix}
           y_1(T_0) \\
           \vdots \\
           y_{m-1}(T_0)
          \end{pmatrix},
\end{equation}

$$ y_1(-0) = z_1, \: \dots \: , \: y_{m-1}(-0) = z_{m-1}, $$
$$ T_0 = \alpha + 1 + (\beta+1)/(\alpha - \beta - 1). $$
\pause
\begin{block}{Теорема}
Любой грубой неподвижной точке $z_{*}$ отображения \eqref{phi_z}, в системе (1) соответствует цикл периода $ T_0 $ с теми же свойствами устойчивости.
\end{block}

\end{frame}

\begin{frame}
\frametitle{Алгоритмическая часть: метод Рунге-Кутты} 
$$
y_j(t+h) = y_j(t) + \frac{h}{6}\:(k_1 +2k_2 + 2k_3 + k_4), \quad j = \overline{1, m-1}\;,
$$

$$
\begin{array}{l}
k_1 = d ( \exp(y_{j+1}) + \exp(-y_j) - \exp(y_j) - \exp(-y_{j-1}) ) ,\\
k_2 = d ( \exp(y_{j+1} + \frac{h}{2}k_1) + \exp(-y_j + \frac{h}{2}k_1) - \exp(y_j + \frac{h}{2}k_1) - \\  \qquad - \exp(-y_{j-1} + \frac{h}{2}k_1) ) ,\\
k_3 = d ( \exp(y_{j+1} + \frac{h}{2}k_2) + \exp(-y_j + \frac{h}{2}k_2) - \exp(y_j + \frac{h}{2}k_2) - \\  \qquad - \exp(-y_{j-1} + \frac{h}{2}k_2) ) ,\\
k_4 = d ( \exp(y_{j+1} + hk_3) + \exp(-y_j + hk_3) - \exp(y_j + hk_3) - \\ \qquad - \exp(-y_{j-1} + hk_3) ),\\
\end{array}
$$

\pause
$$ y_0(t+h) = y_m(t+h) = -\sum\limits_{k=1}^{m-1} y_k(t).  $$
\end{frame}
