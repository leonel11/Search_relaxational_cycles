31  int dim = config.Dim();
32  float d = config.D();
33  float alpha = config.Alpha();
34  float beta = config.Beta();
35  cout << "Stationary points of the theory for given initial values:" << endl;
36  float theor_sln[MAX_DIM]; // массив для хранения координат решения, получающегося из теории
37  PointOperations::ZeroizePoint(theor_sln);
38  cout << TypesChange::PointToString(theor_sln, 1, dim) << endl;
39  for (int k = 0; k <= dim; ++k) {
40  PointOperations::ZeroizePoint(theor_sln);
41  // расчет различных теоретических результатов в зависимости от области параметров
42  // (см. Теорему 2 из файла DU0155.pdf)
43  for (int idx = 1; idx <= dim; ++idx) {
44  // (alpha, beta) in U1 = {(alpha, beta): alpha > (beta+1)*(beta+2), beta > 0} or
45  // (alpha, beta) in U3 = {(alpha, beta): 2*beta+2 < alpha < (beta+1)*(beta+2), beta > 0} or
46  // (alpha, beta) in U4 = {(alpha, beta): max(2, 1+beta) < alpha < 2*beta+2, beta > 0}
47  if ((beta > 0.0 && alpha > (beta+1.0)*(beta+2.0)) ||
48  (beta > 0.0 && alpha > 2.0*beta+2.0 && alpha < (beta+1.0)*(beta+2.0)) ||
49  (beta > 0.0 && alpha > max(2.0, 1.0+beta) && alpha < 2.0*beta+2.0)) {
50  if (idx <= k) // idx = 1...k
51  theor_sln[idx] = log(1.0/d)-log((beta+1.0)/(alpha-beta-1.0))+log((double)k+1.0-idx);
52  else // idx = k+1...m-1
53  theor_sln[idx] = -log(1.0/d)+log((beta+1.0)/(alpha-beta-1.0))-log((double)idx-k);
54  }
55  // (alpha, beta) in U2 = {(alpha, beta): 1 < alpha < 2, 0 < beta<(alpha-1)*(2-alpha)} or
56  // (alpha, beta) in U5 = {(alpha, beta): 1 < alpha < 2, (alpha-1)*(2-alpha) < beta < alpha-1}
57  else if ((alpha > 1.0 && alpha < 2.0 && beta > 0.0 && beta < (alpha-1.0)*(2.0-alpha)) ||
58  (alpha > 1.0 && alpha < 2.0 && beta > (alpha-1.0)*(2.0-alpha) && beta < (alpha-1.0))) {
59  if (idx <= k) // idx = 1...k
60  theor_sln[idx] = -(alpha-1.0)*log(1.0/d)-(alpha-1.0)*log((double)k+1.0-idx)+alpha*log(alpha-1.0);
61  else // idx = k+1...m-1
62  theor_sln[idx] = (alpha-1.0)*log(1.0/d)+(alpha-1.0)*log((double)idx-k)-alpha*log(alpha-1.0);
63  }
64  else {
65  cout << endl;
66  return;
67  }
68  }
69  cout << TypesChange::PointToString(theor_sln, 1, dim) << endl;
70  }
71  cout << endl;
